# ServerCore
It is a Pocketmine plugin with many cool things for example /day make day in the world where you are

# Commands
- /day
- /night
- /heal
- /feed
- /gm 0
- /gm 1
- /gm 2
- /gm 3

# Permissions
- /day      =    day.command.sc
- /night    =    night.command.sc
- /heal     =    heal.command.sc
- /feed     =    feed.command.sc
- /gm 0     =    gm.command.sc
- /gm 1     =    gm.command.sc
- /gm 2     =    gm.command.sc
- /gm 3     =    gm.command.sc
